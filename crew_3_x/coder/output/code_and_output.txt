def calculate_series(terms):
    total = 0.0
    for n in range(terms):
        term = 1 / (2 * n + 1)
        if n % 2 == 0:
            total += term  # Add for even indices
        else:
            total -= term  # Subtract for odd indices
    return total * 4

# Calculate the total for the first 10,000 terms
result = calculate_series(10000)
print(result)
